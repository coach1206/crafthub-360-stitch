MOCK API SOURCE LAYER
=====================

Create:

src/api/passportScanApi.js

Functions:

scanPassportSource(qrPayload)
validateQrPayload(qrPayload)
getVenueById(id)
getEventById(id)
getMemberById(id)
getStampById(id)
getBenefitById(id)
claimStamp(stampId, memberId)
verifyConnection(memberId)
checkInToVenue(venueId)
checkInToEvent(eventId)
redeemBenefit(benefitId)

Each function should return a Promise with simulated delay.

Example scanPassportSource:

async function scanPassportSource(qrPayload) {
const parsed = validateQrPayload(qrPayload)

switch (parsed.sourceType) {
case "venue":
return getVenueById(parsed.sourceId)
case "event":
return getEventById(parsed.sourceId)
case "member":
return getMemberById(parsed.sourceId)
case "stamp":
return getStampById(parsed.sourceId)
case "benefit":
return getBenefitById(parsed.sourceId)
default:
return {
success: false,
error: "Invalid Passport Code"
}
}
}
