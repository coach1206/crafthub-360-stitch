const API_BASE =
  import.meta.env.VITE_API_BASE_URL || "http://localhost:5000/api";

async function request(path, options = {}) {
  const res = await fetch(`${API_BASE}${path}`, {
    headers: {
      "Content-Type": "application/json",
      ...(options.headers || {})
    },
    ...options
  });

  const json = await res.json();

  if (!res.ok || !json.success) {
    throw new Error(json.message || "API request failed");
  }

  return json.data;
}

export function getLeaderboard(venueId = "grand-lounge") {
  return request(`/ranking/leaderboard/${venueId}`);
}

export function getCurrentUserRank(userId = "user-john-collins") {
  return request(`/ranking/current-user/${userId}`);
}

export function processRankingScan(payload) {
  return request("/ranking/scan", {

export function getTickerFeed(filters = {}) {
  const params = new URLSearchParams(filters).toString();
  return request(`/ticker${params ? `?${params}` : ""}`);
}
